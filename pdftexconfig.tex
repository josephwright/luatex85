% Set pdfTeX parameters for pdf mode (replacing pdftex.cfg file).
% Thomas Esser, 2004. public domain.
\begingroup

  % From LuaTeX v0.85 the marker primitive for PDF output is renamed:
  % use a single name.
  \ifx\outputmode\undefined
  \else
    \let\pdfoutput\outputmode
  \fi
  \global\pdfoutput = 1 %

  \ifx\pdfpageheight\undefined
  \else
    \let\pageheight\pdfpageheight
  \fi
  \ifx\pdfpagewidth\undefined
  \else
    \let\pagewidth\pdfpagewidth
  \fi
  \global\pageheight = 297 true mm %
  \global\pagewidth  = 210 true mm %

  % From LuaTeX v0.85 the various primitive settings are accessed using
  % a new mechanism: align this with the pdfTeX one.
  \catcode`\{=1 %
  \catcode`\}=2 %
  \catcode`\#=6 %
  \ifx\pdfvariable\undefined
    \def\pdfsetvar#1 ={\expandafter\global\csname pdf#1\endcsname=}
  \else
    \def\pdfsetvar{\global\pdfvariable}
  \fi
  \pdfsetvar compresslevel = 9 %
  \pdfsetvar decimaldigits = 3 %
  \pdfsetvar horigin       = 1 true in %
  \pdfsetvar vorigin       = 1 true in %
  \pdfsetvar pkresolution  = 600 %
  %
  % As of TeX Live 2010, we output PDF 1.5 by default, so we can enable
  % more compression.  To change this for the installation, comment out or
  % delete these lines and remake the formats.  To change it for a
  % particular LaTeX document, \RequirePackage{pdf14} early.
  %
  \pdfsetvar minorversion     = 5 %
  \pdfsetvar objcompresslevel = 2 %
\endgroup

\endinput
